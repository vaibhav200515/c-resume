5.Write a C++ program to create a class known as "BankAccount" with methods called deposit() and withdraw(). Create a derived class called SavingsAccount that overrides the withdraw() method to prevent withdrawals if the account balance falls below one hundred.

        #include<iostream>
        using namespace std;
        class BankAccount
        {
            public:
                BankAccount()
                {
                    balance=0;
                }
                float money;
                float balance;
                void deposit()
                {
                    cout<<"Input amount:";
                    cin>>money;
                    balance=balance+money;
                    cout<<"Deposited "<<money<<" Successfully!"<<endl;
                    cout<<"Current balance:"<<balance<<endl;
                }
                void withdraw()
                {
                    cout<<"Input amount:";
                    cin>>money;
                    balance=balance-money;
                    cout<<"Withdrawed "<<money<<" Successfully!"<<endl;
                    cout<<"Current balance:"<<balance<<endl;
                }
        };
        class savingsaccount:public BankAccount
        {
            public:
                void withdraw()
                {
                    int money;
                    cout<<"Input amount:";
                    cin>>money;
                    if(balance>100)
                    {
                        balance=balance-money;
                        cout<<"Withdrawed "<<money<<" Successfully!"<<endl;
                        cout<<"Current balance:"<<balance<<endl;
                    }
                    else
                    {
                        cout<<"Can't Withdraw Balance less than 100"<<endl;
                        cout<<"Current balance:"<<balance<<endl;
                    }
                }
        };
        int main()
        {
            savingsaccount s1;
            int x;
            int end;
            for(int i=0;end!=0;i++)
            {
                cout<<"Select Operation:-\n"<<"1.Deposit 2.Withdrawal :";
                cin>>x;
                switch(x)
                {
                    case 1: s1.deposit();
                            break;
                    case 2: s1.withdraw();
                            break;
                    default: cout<<"Invalid Operation!"<<endl;
                            break;
                }
                cout<<"Enter 0 to exit or any other to continue:";
                cin>>end;
            }
        }

8.Write a C++ program to create a class called Shape with methods called getPerimeter() and getArea(). Create a derived class called Circle that overrides the getPerimeter() and getArea() methods to calculate the area and perimeter of a circle.

Source code

    
        #include<iostream>
        #define pi 3.14
        using namespace std;
        class shape
        {
            public:
                float length,breadth,perimeter,area;
                void getperimeter()
                {
                    cout<<"Enter length and breadth:";
                    cin>>length>>breadth;
                    perimeter=2.0*(length+breadth);
                    cout<<"Perimeter:"<<perimeter<<endl;
                }
                void getarea()
                {
                    cout<<"Enter length and breadth:";
                    cin>>length>>breadth;
                    area=length*breadth;
                    cout<<"Area:"<<area<<endl;
                }
        };
        class circle:public shape
        {
            public:
                float r;
                void getperimeter()
                {
                    cout<<"Enter radius:";
                    cin>>r;
                    perimeter=2.0*pi*r;
                    cout<<"Perimeter:"<<perimeter<<endl;
                }
                void getarea()
                {
                    cout<<"Enter radius:";
                    cin>>r;
                    area=pi*r*r;
                    cout<<"Area:"<<area<<endl;
                }
        };
        int main()
        {
            shape s1;
            circle c1;
            s1.getperimeter();
            s1.getarea();
            c1.getperimeter();
            c1.getarea();
        }

Output:

    Enter length and breadth:10 5
    Perimeter:30
    Enter length and breadth:10 5
    Area:50
    Enter radius:12
    Perimeter:75.36
    Enter radius:12
    Area:452.16
